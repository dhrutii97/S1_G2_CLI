\section*{Lecture Topic}
Markov Inequality, Chebyshev Inequality, Chernoff Bound, Jensen Inequality, Tail probabilities, Poisson and Binomial tail analysis

\section{Preliminaries}

\subsection{Tail of Random Variable}
Tail probability measures probability of extreme events:
\[
P(X > x)
\]

Tail represents probability that $X$ takes unusually large values.

\subsection{Why tails matter}
Examples:
\begin{itemize}
\item Jobs delayed more than 24 hours
\item Packets dropped due to full buffer
\end{itemize}

\textbf{Key Insight:}
\begin{itemize}
\item Mean $\rightarrow$ normal behaviour
\item Tail $\rightarrow$ rare extreme events
\end{itemize}

\textbf{Difficulty:} Tail probability often requires summation of many small probabilities.

\section{Model Assumptions and Setup}

We study probability of rare events:
\[
P(X \ge k)
\]

\subsection{Binomial Distribution}
\[
X \sim \text{Binomial}(n,p)
\]

\[
P(X \ge k) = \sum_{i=k}^{n} \binom{n}{i} p^i (1-p)^{n-i}
\]

\textbf{Example:} Distributing $n$ jobs across $n$ servers.

\subsection{Poisson Distribution}
\[
X \sim \text{Poisson}(\lambda)
\]

\[
P(X \ge k) = \sum_{i=k}^{\infty} e^{-\lambda} \frac{\lambda^i}{i!}
\]

\textbf{Example:} Jobs arriving at datacenter.

\subsection{Why exact computation is hard}
\begin{itemize}
\item Long summations
\item Computationally expensive for large $n$
\end{itemize}

\section{Core Results}

\subsection{Markov Inequality}
For non-negative $X$:
\[
P(X \ge a) \le \frac{\mathbb{E}[X]}{a}
\]

\subsection{Chebyshev Inequality}
\[
P(|X - \mu| \ge k\sigma) \le \frac{1}{k^2}
\]

Alternative form:
\[
P(|X - \mu| \ge a) \le \frac{\mathrm{Var}(X)}{a^2}
\]

\subsection{Chernoff Bound}
\[
P(X \ge (1+\delta)\mu) \le \left(\frac{e^\delta}{(1+\delta)^{1+\delta}}\right)^\mu
\]

\subsection{Jensen Inequality}

For convex $f$:
\[
f(\mathbb{E}[X]) \le \mathbb{E}[f(X)]
\]

For concave $f$:
\[
f(\mathbb{E}[X]) \ge \mathbb{E}[f(X)]
\]

\section{Analytical Development}

\subsection{Proof of Markov Inequality}

\[
\mathbb{E}[X] = \sum_x x P(X=x)
\]

\[
\mathbb{E}[X] \ge \sum_{x \ge a} x P(X=x)
\]

\[
\mathbb{E}[X] \ge a \sum_{x \ge a} P(X=x)
\]

\[
\mathbb{E}[X] \ge a P(X \ge a)
\]

\[
P(X \ge a) \le \frac{\mathbb{E}[X]}{a}
\]

\subsection{Proof of Chebyshev Inequality}

Let:
\[
Y = (X - \mu)^2
\]

Applying Markov:
\[
P((X-\mu)^2 \ge a^2) \le \frac{\mathbb{E}[(X-\mu)^2]}{a^2}
\]

\[
\mathbb{E}[(X-\mu)^2] = \mathrm{Var}(X)
\]

\[
P(|X-\mu| \ge a) \le \frac{\mathrm{Var}(X)}{a^2}
\]

\subsection{Chernoff Bound Idea}

\[
P(X \ge a) = P(e^{tX} \ge e^{ta})
\]

\[
P(e^{tX} \ge e^{ta}) \le \frac{\mathbb{E}[e^{tX}]}{e^{ta}}
\]

\subsection{Jensen Inequality Idea}

Using convexity:
\[
f(\mathbb{E}[X]) \le \mathbb{E}[f(X)]
\]

\section{Structural Relationships}

\begin{itemize}
\item Tail probability problem
\item Exact computation difficult
\item Need bounds
\item Markov $\rightarrow$ uses mean
\item Chebyshev $\rightarrow$ uses variance
\item Chernoff $\rightarrow$ exponential bounds
\item Jensen $\rightarrow$ convexity
\end{itemize}

\textbf{Tightness:}
\[
\text{Markov} < \text{Chebyshev} < \text{Chernoff}
\]

\subsection{Key Insights}

\begin{itemize}
\item Tail probability measures rare events
\item Exact computation is expensive
\item Inequalities provide upper bounds
\item Markov uses mean
\item Chebyshev uses variance
\item Chernoff gives tight bounds
\item Jensen relates expectation and convexity
\end{itemize}
