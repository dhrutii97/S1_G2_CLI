\section*{1. Markov’s Inequality}

\subsection*{Statement}

For any non-negative random variable $X$ and $a > 0$:

\[
P(X \ge a) \le \frac{E[X]}{a}
\]

\hrule

\section*{2. Chebyshev’s Inequality}

\subsection*{Statement}

For any random variable $X$ with mean $\mu$ and variance $\sigma^2$, and any $a > 0$:

\[
P(|X - \mu| \ge a) \le \frac{\sigma^2}{a^2}
\]

\hrule

\section*{3. Jensen’s Inequality}

\subsection*{Statement}

For a convex function $g$:

\[
g(E[X]) \le E[g(X)]
\]

\hrule

\section*{4. Problems and Solutions}

\subsection*{Q1}

Let $X \ge 0$ denote annual income with $E[X] = 40000$.

\textbf{(a)}
\[
P(X \ge 120000) \le \frac{40000}{120000} = \frac{1}{3}
\]

\textbf{Result:}
\[
P(X \ge 120000) \le \frac{1}{3}
\]

\textbf{(b)}

Markov’s inequality applies only to probabilities of the form $P(X \ge a)$.

\[
P(X < 10000)
\]

is a lower tail event and cannot be bounded using Markov’s inequality.

\subsection*{Q2}

Let $E[X] = \mu = 100$, $\sigma = 2$, $\sigma^2 = 4$.

Defective condition:
\[
|X - \mu| \ge 6
\]

\textbf{(a)}
\[
P(|X - 100| \ge 6) \le \frac{4}{36} = \frac{1}{9}
\]

\textbf{Result:}
\[
P(\text{defective}) \le \frac{1}{9}
\]

\textbf{(b)}
\[
10000 \times \frac{1}{9} = \frac{10000}{9} \approx 1111
\]

\textbf{Result:} Maximum expected defective $\approx 1111$.

\subsection*{Q3}

Let $0 \le Y \le 1$, $E[Y] = \mu$.

Convex function: $g(x) = e^x$

\textbf{Step 1:}
\[
Y = (1-Y)\cdot 0 + Y \cdot 1
\]

\textbf{Step 2:}
\[
e^{tY} \le (1-Y)e^0 + Y e^t
\]

\[
e^{tY} \le 1 - Y + Y e^t
\]

\textbf{Step 3:}
\[
E[e^{tY}] \le 1 - E[Y] + e^t E[Y]
\]

\[
E[e^{tY}] \le 1 - \mu + \mu e^t
\]

\textbf{Result:}
\[
E[e^{tY}] \le 1 - \mu + \mu e^t
\]

\subsection*{Q4}

Let $X > 0$, $E[X] = \mu$, $g(x) = x^a, a > 1$

\textbf{Step 1:}
\[
g''(x) = a(a-1)x^{a-2} > 0
\]

\textbf{Step 2 (Jensen):}
\[
(E[X])^a \le E[X^a]
\]

\textbf{Result:}
\[
E[X^a] \ge (E[X])^a
\]

\subsection*{Q5}

Let mean $\mu$, variance $\sigma^2$

\textbf{Step 1:} Define $Z = (X-\mu)^2$

\textbf{Step 2:}
\[
P(|X-\mu| \ge k\sigma) = P((X-\mu)^2 \ge k^2\sigma^2)
\]

\textbf{Step 3:}
\[
P((X-\mu)^2 \ge k^2\sigma^2) \le \frac{E[(X-\mu)^2]}{k^2\sigma^2}
\]

\textbf{Step 4:}
\[
E[(X-\mu)^2] = \sigma^2
\]

\[
P(|X-\mu| \ge k\sigma) \le \frac{1}{k^2}
\]

\textbf{Result:}
\[
P(|X-\mu| \ge k\sigma) \le \frac{1}{k^2}
\]

\subsection*{Q6}

\textbf{(a)}
\[
P(X \ge 1000) \le \frac{500}{1000} = \frac{1}{2}
\]

\textbf{(b)}
\[
P(|X - 500| \ge 100) \le \frac{100}{10000} = \frac{1}{100}
\]

\[
P(|X - 500| < 100) \ge \frac{99}{100}
\]

\textbf{Result:} Probability between 400 and 600 $\ge 0.99$

\subsection*{Q7}

\[
E[e^{tX_i}] = \frac{e^t + e^{-t}}{2}
\]

Maclaurin expansions:

\[
e^t = 1 + t + \frac{t^2}{2!} + \cdots
\]

\[
e^{-t} = 1 - t + \frac{t^2}{2!} - \cdots
\]

\[
e^t + e^{-t} = 2\sum_{n=0}^{\infty} \frac{t^{2n}}{(2n)!}
\]

Using:
\[
(2n)! \ge n!2^n
\]

\[
\frac{t^{2n}}{(2n)!} \le \frac{(t^2/2)^n}{n!}
\]

\[
e^t + e^{-t} \le 2e^{t^2/2}
\]

\[
E[e^{tX_i}] \le e^{t^2/2}
\]

For $S_n$:

\[
E[e^{tS_n}] = \prod E[e^{tX_i}] \le (e^{t^2/2})^n = e^{nt^2/2}
\]

\textbf{Result:}
\[
E[e^{tX_i}] \le e^{t^2/2}, \quad E[e^{tS_n}] \le e^{nt^2/2}
\]

\subsection*{Q8}

Define:
\[
X = \sum_{i=1}^{100} X_i
\]

\[
E[X_i] = \frac{100}{199}
\]

\[
E[X] = 100 \cdot \frac{100}{199} \approx 50.25
\]

Variance:
\[
Var(X_i) = \frac{100}{199} \cdot \frac{99}{199}
\]

\[
Cov(X_i, X_j) = \frac{100}{199} \cdot \frac{99}{197} - \left(\frac{100}{199}\right)^2
\]

\[
Var(X) \approx 25.126
\]

Chebyshev:
\[
P(X \le 30) \le \frac{25.126}{(20.25)^2} \approx 0.061
\]

One-sided:
\[
P(X \le 30) \le \frac{25.126}{25.126 + (20.25)^2} \approx 0.058
\]

\subsection*{Q9}

\[
\frac{1}{n}\sum_{i=1}^n \sqrt{i^2+1} \ge \sqrt{1 + \left(\frac{1+2+\cdots+n}{n}\right)^2}
\]

\[
1+2+\cdots+n = \frac{n(n+1)}{2}
\]

\[
\ge \frac{n}{2}\sqrt{n^2 + 2n + 5}
\]

\subsection*{Q10}

Let $X \sim \text{Poisson}(20)$

\[
E[X] = 20, \quad Var(X) = 20
\]

\textbf{(a) Markov:}
\[
P(X \ge 26) \le \frac{20}{26} = \frac{10}{13}
\]

\textbf{(b) One-sided Chebyshev:}
\[
P(X \ge 26) \le \frac{20}{20+36} = \frac{5}{14}
\]

\textbf{(c) Chernoff:}

\[
P(X \ge 26) \le \inf_{t>0} \exp(20(e^t -1) - 26t)
\]

Optimal:
\[
t = \ln(1.3)
\]

\[
P(X \ge 26) \le e^{6 - 26\ln(1.3)} \approx 0.439
\]

\section*{Final Summary}

\begin{itemize}
\item Markov provides loose bounds for non-negative variables.
\item Chebyshev bounds deviation using variance.
\item Jensen connects convexity with expectations.
\item Chernoff gives tighter exponential bounds.
\end{itemize}
